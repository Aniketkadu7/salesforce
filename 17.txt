public class EmployeeManagementSystem {

   // Insert Employee
   public static void addEmployee(Integer empId, String name,
                                  String email, Date birthDate,
                                  String dept) {

       Employee__c emp = new Employee__c();

       emp.Name = name;
       emp.Emp_ID__c = empId;
       emp.Email__c = email;
       emp.Birth_Date__c = birthDate;
       emp.Department__c = dept;

       insert emp;

       System.debug('Employee Added Successfully');
   }

   // View Employees
   public static void viewEmployees() {

       List<Employee__c> empList = [
           SELECT Name, Emp_ID__c, Email__c,
                  Birth_Date__c, Department__c
           FROM Employee__c
       ];

       for(Employee__c e : empList) {

           System.debug(
               'ID: ' + e.Emp_ID__c +
               ' Name: ' + e.Name +
               ' Email: ' + e.Email__c +
               ' Birth Date: ' + e.Birth_Date__c +
               ' Department: ' + e.Department__c
           );
       }
   }

   // Delete Employee
   public static void deleteEmployee(Integer empId) {

       Employee__c emp = [
           SELECT Id
           FROM Employee__c
           WHERE Emp_ID__c = :empId
           LIMIT 1
       ];

       delete emp;

       System.debug('Employee Deleted');
   }
}
